# Enhanced panel function with consistent sizing and better spacing
create_enhanced_panel <- function(boundary, points, title, highlight_state, inset_position = "bottom_right") {
  
  # Ensure same CRS
  boundary <- st_transform(boundary, crs = 4326)
  points <- st_transform(points, crs = 4326)
  
  # Fix validity check
  if (!all(st_is_valid(boundary))) {
    boundary <- st_make_valid(boundary)
  }
  
  # Union boundary if it has multiple features
  if (nrow(boundary) > 1) {
    boundary <- st_union(boundary)
    boundary <- st_sf(geometry = boundary)
  }
  
  # Filter points within boundary
  boundary_buffer <- st_buffer(boundary, dist = 0.01)
  nearby_points <- st_filter(points, boundary_buffer)
  
  # Get tight bounding box around polygon for better zoom
  polygon_bbox <- st_bbox(boundary)
  
  # Calculate ranges for consistent scaling
  x_range <- polygon_bbox[3] - polygon_bbox[1]
  y_range <- polygon_bbox[4] - polygon_bbox[2]
  
  # Make the plot area square-ish and ensure adequate white space
  max_range <- max(x_range, y_range)
  
  # Center the polygon and create consistent plot bounds
  center_x <- (polygon_bbox[1] + polygon_bbox[3]) / 2
  center_y <- (polygon_bbox[2] + polygon_bbox[4]) / 2
  
  # Create consistent plot bounds (square) with enough margin
  margin_factor <- 0.4  # 40% margin around polygon
  half_size <- (max_range * (1 + margin_factor)) / 2
  
  plot_bbox <- c(
    xmin = center_x - half_size,
    ymin = center_y - half_size,
    xmax = center_x + half_size,
    ymax = center_y + half_size
  )
  
  # Set inset coordinates based on position
  if (inset_position == "bottom_right") {
    inset_left <- 0.72
    inset_bottom <- 0.08
    inset_right <- 0.92
    inset_top <- 0.28
  } else { # top_right
    inset_left <- 0.72
    inset_bottom <- 0.72
    inset_right <- 0.92
    inset_top <- 0.92
  }
  
  # Main plot with consistent sizing
  main_plot <- ggplot() +
    geom_sf(data = boundary, fill = "lightblue", alpha = 0.7, 
            color = "black", linewidth = 0.3) +
    geom_sf(data = nearby_points, color = "brown", size = 2, alpha = 0.9) +  # Slightly larger for visibility
    annotation_north_arrow(location = "tl", which_north = "true", 
                          pad_x = unit(0.2, "in"), pad_y = unit(0.2, "in"),  # Even more padding
                          height = unit(0.7, "cm"), width = unit(0.7, "cm"),
                          style = north_arrow_fancy_orienteering(
                            fill = c("black", "white"),
                            line_col = "black",
                            text_family = "Century Gothic")) +
    annotation_scale(location = "bl", width_hint = 0.2,  # Smaller scale bar
                    pad_x = unit(0.2, "in"), pad_y = unit(0.2, "in"),  # More padding
                    height = unit(0.12, "cm"),
                    text_family = "Century Gothic", text_col = "black") +
    coord_sf(xlim = c(plot_bbox[1], plot_bbox[3]), 
             ylim = c(plot_bbox[2], plot_bbox[4]), 
             expand = FALSE) +
    theme_void() +
    theme(
      plot.title = element_text(hjust = 0.5, size = 12, face = "bold", 
                               family = "Century Gothic", margin = margin(b = 10)),
      panel.border = element_rect(color = "black", fill = NA, linewidth = 1),
      plot.margin = margin(5, 5, 5, 5),
      panel.background = element_rect(fill = "white"),
      aspect.ratio = 1  # Force square aspect ratio
    ) +
    labs(title = title)
  
  # Create inset with black dot
  boundary_centroid <- st_centroid(boundary)
  coords <- st_coordinates(boundary_centroid)
  
  inset <- ggplot() +
    geom_sf(data = northeast, fill = "white", color = "gray50", linewidth = 0.2) +
    geom_sf(data = northeast[northeast$name == highlight_state, ], 
            fill = "lightblue", color = "blue", linewidth = 0.3) +
    geom_point(aes(x = coords[1], y = coords[2]), color = "black", size = 1.5) +
    theme_void() +
    theme(
      panel.background = element_rect(fill = "white", color = "black", linewidth = 0.5),
      plot.margin = margin(1, 1, 1, 1)
    ) +
    coord_sf(expand = FALSE)
  
  # Combine with inset
  final_plot <- main_plot + 
    inset_element(inset, left = inset_left, bottom = inset_bottom, 
                  right = inset_right, top = inset_top)
  
  return(final_plot)
}

acadia_plot <- function(boundary, points, title = "Acadia National Park") {
  
  # Ensure same CRS
  boundary <- st_transform(boundary, crs = 4326)
  points <- st_transform(points, crs = 4326)
  
  # Filter points near boundary
  boundary_buffer <- st_buffer(boundary, dist = 0.01)
  nearby_points <- st_filter(points, boundary_buffer)
  
  # Get bounding box with adequate margin
  polygon_bbox <- st_bbox(boundary)
  x_range <- polygon_bbox[3] - polygon_bbox[1]
  y_range <- polygon_bbox[4] - polygon_bbox[2]
  
  # Add 50% margin to ensure no overlap
  expanded_bbox <- c(
    xmin = polygon_bbox[1] - (x_range * 0.3),
    ymin = polygon_bbox[2] - (y_range * 0.3),
    xmax = polygon_bbox[3] + (x_range * 0.3),
    ymax = polygon_bbox[4] + (y_range * 0.3)
  )
  
  # Main plot
  ggplot() +
    geom_sf(data = boundary, fill = "lightblue", alpha = 0.7, 
            color = "black", linewidth = 0.3) +
    geom_sf(data = nearby_points, color = "brown", size = 2, alpha = 0.9) +
